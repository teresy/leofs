      ## disk - raised error times

      ## Threshold disk util% - defalut:100%
      watchdog.disk.threshold_disk_util = 100

      watchdog.disk.threshold_disk_rkb = 262144

      ## Threshold disk write kb/sec - defalut:262144(KB)
